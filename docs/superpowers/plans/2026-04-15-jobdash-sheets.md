# JobDash Google Sheets Tracker — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a fully automated Google Sheets job application tracker with sidebar forms, per-job detail tabs, analytics charts, Sankey diagram, and Google Drive integration for JD/Q&A storage.

**Architecture:** Single Google Spreadsheet with bound Apps Script. HTML sidebar forms for data entry. Google Drive folder structure for rich-text documents (JD, form answers). Native Sheets charts for analytics plus Google Charts API Sankey in a modal dialog. All automation via onEdit and daily time-driven triggers.

**Tech Stack:** Google Apps Script, Google Sheets API, Google Drive API, HTML Service (sidebar/dialog), Google Charts API (Sankey), clasp CLI (local development)

**Spec:** `docs/superpowers/specs/2026-04-15-jobdash-sheets-design.md`

---

## Development Workflow Note

Google Apps Script has no standard test framework. Instead of TDD, each task follows:
1. Write the code locally
2. Push to Google via `clasp push`
3. Verify behavior manually in the spreadsheet
4. Commit locally

All `.gs` and `.html` files live in a `src/` directory locally and get pushed to the bound Apps Script project.

// src/Utils.gs

/**
 * Gets the _Config sheet.
 */
function getConfigSheet() {
  return SpreadsheetApp.getActiveSpreadsheet().getSheetByName('_Config');
}

/**
 * Gets the Dashboard sheet.
 */
function getDashboardSheet() {
  return SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Dashboard');
}

/**
 * Reads a config list from _Config tab.
 * Starts reading from the row AFTER the header label, stops at first empty cell.
 * @param {string} headerLabel - e.g. "STATUSES", "SOURCES"
 * @param {number} col - column number (1-indexed)
 * @returns {string[]}
 */
function getConfigList(headerLabel, col) {
  const sheet = getConfigSheet();
  const data = sheet.getRange(1, col, sheet.getLastRow(), 1).getValues();
  let startIdx = -1;
  for (let i = 0; i < data.length; i++) {
    if (data[i][0] === headerLabel) {
      startIdx = i + 1;
      break;
    }
  }
  if (startIdx === -1) return [];
  const result = [];
  for (let i = startIdx; i < data.length; i++) {
    if (data[i][0] === '' || data[i][0] === null) break;
    result.push(data[i][0]);
  }
  return result;
}

/**
 * Gets status list from config.
 */
function getStatuses() {
  return getConfigList('STATUSES', 1);
}

/**
 * Gets status color mapping: { status: { bg, text, sort } }
 */
function getStatusColors() {
  const sheet = getConfigSheet();
  const data = sheet.getRange(1, 1, sheet.getLastRow(), 4).getValues();
  let startIdx = -1;
  for (let i = 0; i < data.length; i++) {
    if (data[i][0] === 'STATUSES') {
      startIdx = i + 1;
      break;
    }
  }
  if (startIdx === -1) return {};
  const colors = {};
  for (let i = startIdx; i < data.length; i++) {
    if (data[i][0] === '' || data[i][0] === null) break;
    colors[data[i][0]] = {
      bg: data[i][1],
      text: data[i][2],
      sort: parseInt(data[i][3], 10)
    };
  }
  return colors;
}

/**
 * Formats a Date object as "MMM DD, YYYY".
 */
function formatDate(date) {
  if (!date) return '';
  const d = new Date(date);
  const months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
  return months[d.getMonth()] + ' ' + d.getDate() + ', ' + d.getFullYear();
}

/**
 * Formats a Date as "MMM DD" (short form for dashboard).
 */
function formatDateShort(date) {
  if (!date) return '';
  const d = new Date(date);
  const months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
  return months[d.getMonth()] + ' ' + d.getDate();
}

/**
 * Generates a unique tab name for a job. Appends (2), (3) etc. if duplicate.
 * @param {string} company
 * @param {string} role
 * @returns {string}
 */
function generateTabName(company, role) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheets = ss.getSheets().map(s => s.getName());
  let base = company + ' - ' + role;
  if (!sheets.includes(base)) return base;
  let counter = 2;
  while (sheets.includes(base + ' (' + counter + ')')) {
    counter++;
  }
  return base + ' (' + counter + ')';
}

/**
 * Finds the Dashboard row (1-based) for a job tab by matching company (col A) + role (col B).
 * @param {string} tabName - e.g. "Google - SDE2" or "Google - SDE2 (2)"
 * @returns {number|null} - row number or null
 */
function findDashboardRowByTab(tabName) {
  const sheet = getDashboardSheet();
  const lastRow = sheet.getLastRow();
  if (lastRow < 3) return null;
  const data = sheet.getRange(3, 1, lastRow - 2, 2).getValues(); // cols A+B
  // Parse company and role from tab name (strip trailing " (N)" suffix)
  const cleaned = tabName.replace(/\s*\(\d+\)$/, '');
  const sepIdx = cleaned.indexOf(' - ');
  const company = sepIdx >= 0 ? cleaned.substring(0, sepIdx) : cleaned;
  const role = sepIdx >= 0 ? cleaned.substring(sepIdx + 3) : '';
  for (let i = 0; i < data.length; i++) {
    if (data[i][0] === company && data[i][1] === role) return i + 3;
  }
  return null;
}

/**
 * Gets all active job tab names (excludes Dashboard, Analytics, _Template, _Config).
 */
function getJobTabNames() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const reserved = ['Dashboard', 'Analytics', '_Template', '_Config'];
  return ss.getSheets()
    .map(s => s.getName())
    .filter(n => !reserved.includes(n));
}

/**
 * HTML include helper — loads an HTML file's content for templating.
 * Used as: <?!= include('Styles') ?>
 */
function include(filename) {
  return HtmlService.createHtmlOutputFromFile(filename).getContent();
}

/**
 * Installable onEdit trigger handler.
 * Updates Last Updated, tracks status changes, logs activity.
 */
function onEditHandler(e) {
  const sheet = e.source.getActiveSheet();
  const range = e.range;

  // Only handle Dashboard edits
  if (sheet.getName() !== 'Dashboard') return;

  const row = range.getRow();
  // Skip header rows (1=stats, 2=headers)
  if (row < 3) return;

  const col = range.getColumn();

  // Update Last Updated (column 9) on any edit
  sheet.getRange(row, 9).setValue(new Date());

  // If Status column (3) changed, handle status-specific logic
  if (col === 3) {
    const newStatus = range.getValue();
    const oldStatus = e.oldValue || '';

    // Update Status Changed Date (hidden column 17)
    sheet.getRange(row, 17).setValue(new Date());

    // Reset Days in Status to 0
    sheet.getRange(row, 11).setValue(0);

    // Auto-set Applied Date if transitioning from Wishlist
    if (oldStatus === 'Wishlist' && newStatus !== 'Wishlist') {
      sheet.getRange(row, 8).setValue(new Date());
    }

    // Update status cell color
    const statusColors = getStatusColors();
    const color = statusColors[newStatus];
    if (color) {
      range.setBackground(color.bg).setFontColor(color.text);
    }

    // Update tab color
    const company = sheet.getRange(row, 1).getValue();
    const role = sheet.getRange(row, 2).getValue();
    const tabName = findJobTab(company, role);
    if (tabName) {
      const jobSheet = e.source.getSheetByName(tabName);
      if (jobSheet && color) {
        jobSheet.setTabColor(color.bg);
      }

      // Log status change in Activity Log
      appendActivityLog(jobSheet, oldStatus + ' → ' + newStatus);
    }

    // If changing to concluded status, un-grey Post-Process
    if (['Accepted', 'Rejected', 'Withdrawn'].includes(newStatus)) {
      if (tabName) {
        const jobSheet = e.source.getSheetByName(tabName);
        // Un-grey post-process rows (40-45)
        if (jobSheet) {
          jobSheet.getRange('A40:E45').setFontColor('#555555');
        }
      }
    }
  }
}

/**
 * Finds the job tab name by matching company and role.
 * Looks for "{company} - {role}" pattern.
 */
function findJobTab(company, role) {
  const tabNames = getJobTabNames();
  const target = company + ' - ' + role;
  // Exact match first
  if (tabNames.includes(target)) return target;
  // Try with number suffix
  for (const name of tabNames) {
    if (name.startsWith(target)) return name;
  }
  return null;
}

/**
 * Appends an entry to the Activity Log section of a job tab.
 * Activity Log starts at row 48.
 */
function appendActivityLog(jobSheet, description) {
  // Find the next empty row in the activity log area (starting row 48)
  const logStart = 48;
  const lastRow = jobSheet.getLastRow();
  const nextRow = Math.max(logStart, lastRow + 1);

  jobSheet.getRange(nextRow, 1).setValue(formatDateShort(new Date()));
  jobSheet.getRange(nextRow, 2).setValue(description);
  jobSheet.getRange(nextRow, 1).setFontColor('#bbbbbb');
  jobSheet.getRange(nextRow, 2).setFontColor('#999999');
}

/**
 * Installs the onEdit trigger. Run this once manually.
 */
function installTriggers() {
  // Remove existing onEdit triggers to avoid duplicates
  const triggers = ScriptApp.getProjectTriggers();
  triggers.forEach(function(trigger) {
    if (trigger.getHandlerFunction() === 'onEditHandler') {
      ScriptApp.deleteTrigger(trigger);
    }
  });

  // Install new onEdit trigger
  ScriptApp.newTrigger('onEditHandler')
    .forSpreadsheet(SpreadsheetApp.getActive())
    .onEdit()
    .create();

  SpreadsheetApp.getUi().alert('Triggers installed successfully.');
}

/**
 * Adds an interview round to a job's detail tab.
 * Rounds are inserted starting at row 14 in the right column (D:E).
 * Each round takes 12 rows.
 */
function addInterviewRound(data) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const jobSheet = ss.getSheetByName(data.application);
  if (!jobSheet) throw new Error('Tab not found: ' + data.application);

  // Count existing rounds by scanning for "R" pattern in column D
  let roundNumber = 0;
  const roundStartRow = 14;
  const colD = jobSheet.getRange('D14:D100').getValues();
  for (let i = 0; i < colD.length; i++) {
    if (String(colD[i][0]).match(/^R\d+$/)) roundNumber++;
  }
  roundNumber++;

  // Calculate insertion row
  const insertRow = roundStartRow + ((roundNumber - 1) * 13);

  // Insert rows if needed
  if (insertRow > jobSheet.getMaxRows() - 20) {
    jobSheet.insertRowsAfter(jobSheet.getMaxRows(), 20);
  }

  const roundType = data.roundType === 'Custom' ? data.customTypeName : data.roundType;
  const isScheduled = data.status === 'Scheduled';

  // Write round card data
  jobSheet.getRange(insertRow, 4).setValue('R' + roundNumber);
  jobSheet.getRange(insertRow, 5).setValue(roundType);
  jobSheet.getRange(insertRow, 4).setFontWeight('bold').setFontColor('#2d2d2d').setFontSize(13);
  jobSheet.getRange(insertRow, 5).setFontColor('#888888').setFontSize(12);

  // Status row
  jobSheet.getRange(insertRow + 1, 4).setValue('Status:');
  jobSheet.getRange(insertRow + 1, 5).setValue(data.status + (data.outcome ? '  |  Outcome: ' + data.outcome : ''));
  jobSheet.getRange(insertRow + 1, 4).setFontColor('#999999');

  // Date, duration, interviewer row
  const dateStr = data.date ? formatDateShort(new Date(data.date)) : '';
  const details = [dateStr, data.duration + 'm', data.interviewer].filter(Boolean).join(' · ');
  jobSheet.getRange(insertRow + 2, 4).setValue(details);
  jobSheet.getRange(insertRow + 2, 4).setFontColor('#888888').setFontSize(11);

  // Interviewer LinkedIn
  if (data.interviewerLinkedin) {

    );
    jobSheet.getRange(insertRow + 2, 5).setFontColor('#7b9ec4').setFontSize(11);

    );

  const folder = getOrCreateJobFolder(tabName);

  // Check if doc already exists
  const existingDocs = folder.getFilesByName(tabName + ' — Job Description');
  if (existingDocs.hasNext()) {
    return existingDocs.next().getUrl();
  }

  const doc = DocumentApp.create(tabName + ' — Job Description');

/**
 * Sorts Dashboard by status priority (from _Config), then by deadline urgency.
 */
function sortDashboard() {
  const dashboard = getDashboardSheet();
  const lastRow = dashboard.getLastRow();
  if (lastRow < 3) return;

  const statusColors = getStatusColors();
  const dataRange = dashboard.getRange(3, 1, lastRow - 2, 17);
  const data = dataRange.getValues();

  data.sort(function(a, b) {
    // Column C (index 2) = status, Column J (index 9) = deadline
    const statusA = statusColors[a[2]] ? statusColors[a[2]].sort : 99;
    const statusB = statusColors[b[2]] ? statusColors[b[2]].sort : 99;
    if (statusA !== statusB) return statusA - statusB;

    // Secondary sort: deadline (earliest first, empty last)
    const deadlineA = a[9] ? new Date(a[9]).getTime() : Infinity;
    const deadlineB = b[9] ? new Date(b[9]).getTime() : Infinity;
    return deadlineA - deadlineB;
  });

  dataRange.setValues(data);

  // Re-apply status colors after sort
  for (let i = 0; i < data.length; i++) {
    const status = data[i][2];
    const color = statusColors[status];
    if (color) {
      dashboard.getRange(i + 3, 3).setBackground(color.bg).setFontColor(color.text);
    }
  }

  // Re-apply Details hyperlinks (formulas get lost on setValues)
  reapplyDetailsLinks();

  SpreadsheetApp.getActiveSpreadsheet().toast('Dashboard sorted.', 'JobDash');
}

/**

function refreshDaysInStatus() {
  const dashboard = getDashboardSheet();
  const lastRow = dashboard.getLastRow();
  if (lastRow < 3) return;

  const today = new Date();
  today.setHours(0, 0, 0, 0);

  for (let r = 3; r <= lastRow; r++) {
    const statusChangedDate = dashboard.getRange(r, 17).getValue();
    if (statusChangedDate) {
      const changed = new Date(statusChangedDate);
      changed.setHours(0, 0, 0, 0);
      const days = Math.floor((today - changed) / (1000 * 60 * 60 * 24));
      dashboard.getRange(r, 11).setValue(days);
    }
  }
}

/**
 * Refreshes summary stats in row 1 of Dashboard.
 * Also refreshes Analytics tab if it exists.
 */
function refreshStats() {
  const dashboard = getDashboardSheet();
  const lastRow = dashboard.getLastRow();
  if (lastRow < 3) return;

  const data = dashboard.getRange(3, 1, lastRow - 2, 17).getValues();

  // Total Active (not Rejected, Withdrawn, Accepted, or empty)
  const concluded = ['Rejected', 'Withdrawn', 'Accepted'];
  const active = data.filter(row => row[2] && !concluded.includes(row[2]));
  dashboard.getRange('B1').setValue(active.length);

  // Offer Rate
  const applied = data.filter(row => row[2] && row[2] !== 'Wishlist');
  const offers = data.filter(row => ['Offer', 'Accepted'].includes(row[2]));
  const offerRate = applied.length > 0 ? Math.round((offers.length / applied.length) * 100) : 0;
  dashboard.getRange('H1').setValue(offerRate + '%');

  // Interviews This Week — scan job tabs for interview dates in current week
  const today = new Date();
  const weekStart = new Date(today);
  weekStart.setDate(today.getDate() - today.getDay());
  weekStart.setHours(0, 0, 0, 0);
  const weekEnd = new Date(weekStart);
  weekEnd.setDate(weekStart.getDate() + 7);

  let interviewsThisWeek = 0;
  const tabNames = getJobTabNames();
  for (const tabName of tabNames) {
    const jobSheet = ss.getSheetByName(tabName);
    if (!jobSheet) continue;
    // Scan column D for date-like values in interview rounds area (row 14+)
    const lastRow = jobSheet.getLastRow();
    if (lastRow < 14) continue;
    const roundData = jobSheet.getRange(14, 4, lastRow - 13, 2).getValues();
    for (const row of roundData) {
      // Look for rows containing date strings that match current week
      const text = String(row[0]);
      if (text.match(/^R\d+$/)) continue; // Round header
      // Check for date patterns like "Apr 17" in detail rows
      const dateMatch = text.match(/^([A-Z][a-z]{2}\s+\d{1,2})/);
      if (dateMatch) {
        const year = today.getFullYear();
        const parsed = new Date(dateMatch[1] + ', ' + year);
        if (parsed >= weekStart && parsed < weekEnd) {
          interviewsThisWeek++;
        }
      }
    }
  }
  dashboard.getRange('D1').setValue(interviewsThisWeek);

  // Avg Days to Response
  const responseDays = [];
  for (const row of data) {
    if (row[7] && row[16]) { // Applied date and Status Changed Date
      const applied = new Date(row[7]);
      const changed = new Date(row[16]);
      const diff = Math.floor((changed - applied) / (1000 * 60 * 60 * 24));
      if (diff > 0) responseDays.push(diff);
    }
  }
  const avgDays = responseDays.length > 0
    ? Math.round(responseDays.reduce((a, b) => a + b, 0) / responseDays.length)
    : '—';
  dashboard.getRange('F1').setValue(avgDays + (typeof avgDays === 'number' ? 'd' : ''));

  // Refresh analytics if tab exists
  try { refreshAnalytics(); } catch(e) { /* Analytics tab may not exist yet */ }

  SpreadsheetApp.getActiveSpreadsheet().toast('Stats refreshed.', 'JobDash');
}

/**
 * Computes Sankey diagram data: status transition counts.
 * Reads Activity Log from each job tab for "Status: X → Y" entries.
 * @returns {Array} - [[from, to, count], ...]
 */
function getSankeyData() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const tabNames = getJobTabNames();
  const transitions = {};

  for (const tabName of tabNames) {
    const sheet = ss.getSheetByName(tabName);
    if (!sheet) continue;

    // Activity log starts at row 48
    const lastRow = sheet.getLastRow();
    if (lastRow < 48) continue;

    const logData = sheet.getRange(48, 2, lastRow - 47, 1).getValues();
    for (const row of logData) {
      const text = String(row[0]);
      // Match "Status: X → Y" or "X → Y"
      const match = text.match(/(?:Status:\s*)?(.+?)\s*→\s*(.+)/);
      if (match) {
        const from = match[1].trim();
        const to = match[2].trim();
        const key = from + '|' + to;
        transitions[key] = (transitions[key] || 0) + 1;
      }
    }
  }

  // Convert to array format
  const result = [];
  for (const key in transitions) {
    const parts = key.split('|');
    result.push([parts[0], parts[1], transitions[key]]);
  }

  return result;
}

function installTriggers() {
  const triggers = ScriptApp.getProjectTriggers();

  // Remove existing managed triggers
