       export const ANALYTICS_THRESHOLD_DEFAULTS = {
         appliedScreening:      { greenBelow: 7, amberBelow: 14 },

         interviewingOffer:     thresholdPair,

             && t.screeningInterviewing.greenBelow < t.screeningInterviewing.amberBelow
