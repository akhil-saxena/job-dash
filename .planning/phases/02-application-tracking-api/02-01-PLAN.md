export function generateBaseSlug(company: string, role: string): string {
  return `${company}-${role}`
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, "-")
    .replace(/^-|-$/g, "");
}

Create `application` table following the exact same Drizzle patterns as auth.ts:
- `id`: text PK with nanoid() default

- `companyName`: text, notNull
- `roleTitle`: text, notNull
- `jobPostingUrl`: text, nullable

- `locationCity`: text, nullable
- `salaryMin`: integer, nullable
- `salaryMax`: integer, nullable
- `salaryOffered`: integer, nullable
- `salaryCurrency`: text, default "INR"
- `equity`: text, nullable
- `bonus`: text, nullable
- `status`: text, notNull, default "wishlist"
- `priority`: text, default "medium"
- `source`: text, nullable
- `isPinned`: integer mode boolean, notNull, default false
- `isArchived`: integer mode boolean, notNull, default false
- `notes`: text, nullable (markdown)
- `slug`: text, notNull
- `appliedAt`: integer mode timestamp, nullable
- `createdAt`: integer mode timestamp, notNull, default sql`(unixepoch())`
- `updatedAt`: integer mode timestamp, notNull, default sql`(unixepoch())`
- `deletedAt`: integer mode timestamp, nullable

Indexes (in the table's third argument array per Drizzle v0.45 API):
- `uniqueIndex("idx_application_user_slug").on(t.userId, t.slug)` -- per D-15
- `index("idx_application_user_status").on(t.userId, t.status)` -- for filtering
- `index("idx_application_user_created").on(t.userId, t.createdAt)` -- for sorting
- `index("idx_application_user_archived").on(t.userId, t.isArchived)` -- for default filter
- `index("idx_application_user_pinned").on(t.userId, t.isPinned)` -- for pinned filter

Create `timelineEvent` table per D-10:
- `id`: text PK with nanoid() default
- `applicationId`: text FK to application.id with cascade delete
- `userId`: text FK to user.id with cascade delete
- `eventType`: text, notNull (values from D-11)
- `description`: text, notNull
- `metadata`: text, nullable (JSON string)
- `occurredAt`: integer mode timestamp, notNull, default sql`(unixepoch())`

Indexes:
- `index("idx_timeline_event_app").on(t.applicationId, t.occurredAt)` -- for timeline queries

IMPORTANT: Do NOT use `.where()` on partial indexes -- Drizzle's SQLite index builder may not support partial index `.where()` clause reliably in v0.45. Use standard composite indexes instead.
