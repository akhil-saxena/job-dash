```typescript
function useDebouncedMutate(mutate: (fields: Record<string, unknown>) => void, delay = 800) {
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
  const debouncedMutate = useCallback((fields: Record<string, unknown>) => {
```

Hooks to create:

**`useInterviews(applicationId: string)`** -- query key: `["interviews", applicationId]`, fetches `GET /api/applications/${applicationId}/interviews`, returns `InterviewRound[]` from `json.data`, `enabled: !!applicationId`.

**`useInterviewCount(applicationId: string)`** -- query key: `["interview-count", applicationId]`, fetches `GET /api/applications/${applicationId}/interview-count`, returns `number` from `json.data.count`, `enabled: !!applicationId`.

**`useCreateRound(applicationId: string)`** -- mutation, POSTs to `/api/applications/${applicationId}/interviews`, on success invalidates `["interviews", applicationId]` and `["interview-count", applicationId]`. Shows toast on error.

**`useUpdateRound(applicationId: string)`** -- mutation accepting `{ roundId: string } & Partial fields`. PATCHes `/api/interviews/${roundId}`. On success invalidates `["interviews", applicationId]`. No toast on success (silent auto-save). Toast on error.

**`useDeleteRound(applicationId: string)`** -- mutation accepting `{ roundId: string }`. DELETEs `/api/interviews/${roundId}`. On success invalidates both queries. Toast "Round deleted" on success.

**`useCreateQA(applicationId: string)`** -- mutation accepting `{ roundId: string, question: string, answer?: string }`. POSTs to `/api/interviews/${roundId}/qa`. Invalidates `["interviews", applicationId]`.

**`useUpdateQA(applicationId: string)`** -- mutation accepting `{ qaId: string, question?: string, answer?: string }`. PATCHes `/api/interview-qa/${qaId}`. Invalidates `["interviews", applicationId]`. Silent (auto-save).

**`useDeleteQA(applicationId: string)`** -- mutation accepting `{ qaId: string }`. DELETEs `/api/interview-qa/${qaId}`. Invalidates `["interviews", applicationId]`.

```typescript
import { Star } from "lucide-react";
import { useState } from "react";

interface StarRatingProps {
  value: number | null;
  onChange?: (rating: number | null) => void;
  readOnly?: boolean;
  size?: number;
}

export function StarRating({ value, onChange, readOnly = false, size = 16 }: StarRatingProps) {
  const [hovered, setHovered] = useState<number | null>(null);
  const display = hovered ?? value ?? 0;

  return (
    <div className="flex gap-0.5" onMouseLeave={() => !readOnly && setHovered(null)}>
      {[1, 2, 3, 4, 5].map((star) => (
        <button
          key={star}
          type="button"
          disabled={readOnly}
          onClick={() => onChange?.(value === star ? null : star)}
          onMouseEnter={() => !readOnly && setHovered(star)}
          className={`${readOnly ? "cursor-default" : "cursor-pointer hover:scale-110"} transition-transform`}
        >
          <Star
            size={size}
            className={star <= display
              ? "fill-amber-400 text-amber-400"
              : "fill-transparent text-black/20 dark:text-white/20"}
          />
        </button>
      ))}
    </div>
  );
}
```

Key behaviors: click fills stars up to clicked position, click same star again clears rating (sets to null), hover shows preview, readOnly disables interaction, amber fill color matches design system.

```typescript
import { useState } from "react";

interface InterviewsTabProps {

        >
          <Plus size={12} /> Add interview

      </div>

      {/* Add round form (inline, toggled) */}

        <div className="rounded-[14px] bg-white/55 backdrop-blur-[14px] border border-amber-200/40 p-4 dark:bg-zinc-800/50 dark:border-amber-500/20">
          <div className="flex items-end gap-3">

            >
              Cancel
            </button>
          </div>
        </div>
      )}

      {/* Round cards */}
      {rounds && rounds.length > 0 ? (
        rounds.map((round, i) => (
          <InterviewRoundCard
            key={round.id}
            round={round}
            index={i}
            applicationId={app.id}
          />
        ))
      ) : (
        <div className="rounded-[14px] bg-white/55 backdrop-blur-[14px] border border-white/50 p-8 text-center dark:bg-zinc-800/50 dark:border-white/10">
          <p className="text-[13px] text-text-muted dark:text-dark-accent/40">
            No interviews scheduled yet. Click "Add interview" to log your first round.
          </p>
        </div>
      )}
    </div>
  );
}
```
