**1. Create `src/client/hooks/useCompany.ts`:**

```typescript
// Company interface
export interface Company {
  id: string;
  name: string;
  domain: string | null;
  website: string | null;
  notes: string | null;
  createdAt: number;
  updatedAt: number;
}

// useFindOrCreateCompany(companyName, domain?) -- POST /api/companies
//   queryKey: ["company", companyName]
//   Uses useMutation, not useQuery -- triggered explicitly when detail page loads
//   Actually, better: useQuery with POST-like behavior. 
//   Approach: Use useQuery that POSTs to findOrCreate on mount.
//   The findOrCreate endpoint is idempotent (returns existing or creates new).

// Better pattern: useCompanyForApplication(companyName)
//   - useQuery that calls POST /api/companies with { name: companyName }
//   - queryKey: ["company", companyName.toLowerCase().trim()]
//   - findOrCreate is idempotent so this is safe as a query
//   - Returns Company object

export function useCompanyForApplication(companyName: string) {
  return useQuery({
    queryKey: ["company", companyName.toLowerCase().trim()],
    queryFn: async () => {
      // Extract domain from company name (simple heuristic: lowercase + .com)
      // This is a best-effort guess; user can set domain manually
      const res = await fetch("/api/companies", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ name: companyName }),
      });
      if (!res.ok) throw new Error("Failed to find/create company");
      const json = await res.json();
      return json.data as Company;
    },
    enabled: !!companyName,
    staleTime: 5 * 60 * 1000, // 5 min -- company doesn't change often
  });
}

// useUpdateCompany() -- PATCH /api/companies/:companyId
//   Invalidates ["company", ...] on success
export function useUpdateCompany() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: async ({ companyId, ...fields }: { companyId: string; [key: string]: unknown }) => {
      const res = await fetch(`/api/companies/${companyId}`, {
        method: "PATCH",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(fields),
      });
      if (!res.ok) throw new Error("Failed to update company");
      return res.json();
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["company"] });
    },
  });
}
```
